SELECT nazwa, dlugosc, szerokosc FROM smok WHERE pochodzenie = "Polska";

SELECT DISTINCT pochodzenie FROM smok GROUP BY pochodzenie ASC;

SELECT parada.rok, AVG(smok.dlugosc) AS "Średnia długość" FROM parada JOIN smok ON parada.id = smok.id WHERE rok > 2005 GROUP BY parada.rok;
ALTER TABLE parada ADD lokalizacja varchar(100);
